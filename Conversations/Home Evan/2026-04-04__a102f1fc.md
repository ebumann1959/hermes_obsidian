      r = self._client.get("/token-boosts/latest/v1")

      if isinstance(data, list):

      return data.get("data") or []

  def fetch_token_boosts(self, chains=None, max_age_hours=24.0,
                         min_liquidity_usd=10_000.0, min_age_minutes=5.0):
      snapshot_ts = datetime.now(timezone.utc)
      boosts = self.token_boosts()
      addresses = []
      for b in boosts:
          if chains and b.get("chainId") not in chains: continue
          addr = b.get("tokenAddress") or b.get("address")
          if addr and addr not in addresses: addresses.append(addr)
      if not addresses: return []
      all_pairs = []
      for i in range(0, len(addresses), _MAX_ADDRESSES_PER_CALL):
          batch = addresses[i: i + _MAX_ADDRESSES_PER_CALL]
          try: all_pairs.extend(self.token_pairs(*batch))
          except Exception: continue
      return _apply_filters(all_pairs, snapshot_ts, chains, max_age_hours,
                            min_liquidity_usd, min_age_minutes)
